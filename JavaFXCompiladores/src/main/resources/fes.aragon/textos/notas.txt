int estado = 0;
    int contador = 0;

    // "While no fin-cadena do"
    while (contador < cadena.length()) {
        char simbolo = cadena.charAt(contador);

        // Primero validamos que el usuario no meta letras o números raros
        if (simbolo != '0' && simbolo != '1') {
            throw new Exception("Error: El alfabeto solo permite '0' y '1'. Símbolo inválido: '" + simbolo + "'");
        }

        // Evaluamos en qué estado estamos (q0, q1, q2, q3, q4)
        switch (estado) {
            case 0: // q0: Esperando el primer '0'
                if (simbolo == '0') estado = 1;
                else throw new Exception("Error: La cadena debe empezar con '00'."); // Reemplaza a qm
                break;

            case 1: // q1: Llevamos un '0', esperando el segundo
                if (simbolo == '0') estado = 2;
                else throw new Exception("Error: La cadena debe empezar con '00'."); // Reemplaza a qm
                break;

            case 2: // q2: Estado de aceptación. Ya vimos "00"
                if (simbolo == '0') estado = 2; // Sigue siendo válido
                else if (simbolo == '1') estado = 3; // Rompió la racha final
                break;

            case 3: // q3: Vimos un '1', necesitamos buscar ceros de nuevo
                if (simbolo == '0') estado = 4;
                else if (simbolo == '1') estado = 3;
                break;

            case 4: // q4: Vimos el primer '0' de la posible racha final
                if (simbolo == '0') estado = 2; // ¡Recuperamos el "00"!
                else if (simbolo == '1') estado = 3; // Falsa alarma, volvemos a buscar
                break;
        }
        contador++;
    }

    // Al terminar de leer toda la palabra, verificamos si nos quedamos en la meta
    if (estado == 2) {
        return true;
    } else {
        // Si terminó en 0, 1, 3 o 4, significa que quedó incompleta
        throw new Exception("Error: La cadena quedó incompleta y no termina con '00'.");
    }
